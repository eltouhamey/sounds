<script>

function playSound(sound) {

var audio = document.getElementById('audio');

if (sound == 'dog') {

audio.src = 'dog.mp3';

} else if (sound == 'cat') {

audio.src = 'cat.mp3';

} else if (sound == 'bird') {

audio.src = 'bird.mp3';

}

audio.play();

}
